find_package(Android)

#

set(CLRSTC_SRC_ROOT ${CMAKE_CURRENT_LIST_DIR}/../src)

#

        ${ANDROID_INCLUDE_DIRS}

        ${CLRSTC_SRC_ROOT})

        ${CLRSTC_SRC_ROOT}/core/and/log_buffer.cpp

        ${CLRSTC_SRC_ROOT}/core/engine.cpp

        ${CLRSTC_SRC_ROOT}/renderer/renderer.cpp)

#

        ${ANDROID_LIBRARIES})
